:param results: JSON Object :return: List of dictionaries [ { “title”: “title of the news”, “link”: “original link of the news source”, “source”:”your-api-name” }, … ]

def newsapi_response_parser(results):
# 
    response = []
    for child in results["articles"]:
        response.append({
            "title": child["title"],
            "link": child["url"],
            "source": "newsapi"
        })
    return response


NEWS_API_MAPPING = {
    "api_name": "newsapi",
